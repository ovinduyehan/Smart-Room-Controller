#include <DHT.h>

#define PIR_PIN 2

#define DHTPIN 3
#define DHTTYPE DHT22

#define LIGHT_PIN 8
#define FAN_PIN 9

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);

  pinMode(PIR_PIN, INPUT);

  pinMode(LIGHT_PIN, OUTPUT);
  pinMode(FAN_PIN, OUTPUT);

  dht.begin();

  Serial.println("Smart Room Controller Started");
}

void loop() {

  // Read sensors
  int motion = digitalRead(PIR_PIN);
  float temperature = dht.readTemperature();
  float humidity = dht.readHumidity();

  // Check if DHT22 reading failed
  if (isnan(temperature) || isnan(humidity)) {
    Serial.println("Failed to read DHT22 sensor!");
    return;
  }

  // Light Control
  if (motion == HIGH) {
    digitalWrite(LIGHT_PIN, HIGH);
    Serial.println("Motion Detected - Light ON");
  } else {
    digitalWrite(LIGHT_PIN, LOW);
    Serial.println("No Motion - Light OFF");
  }

  // Fan Control
  if (temperature > 30) {
    digitalWrite(FAN_PIN, HIGH);
    Serial.println("Temperature High - Fan ON");
  } else {
    digitalWrite(FAN_PIN, LOW);
    Serial.println("Temperature Normal - Fan OFF");
  }

  // Display sensor values
  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.println(" °C");

  Serial.print("Humidity: ");
  Serial.print(humidity);
  Serial.println(" %");

  Serial.println("---------------------------");

  delay(1000);
}
